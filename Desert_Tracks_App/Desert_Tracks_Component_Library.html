        function clamp(x){ return Math.min(0, Math.max(maxScroll(), x)); }

            curX = clamp(x);

            track.style.transform = 'translateX(' + curX + 'px)';
            updateUI();

        function updateUI(){
            var max = -maxScroll();
            var pct = max > 0 ? (-curX / max) * 100 : 0;
            if (progress) progress.style.width = pct + '%';
            
            var pos = -curX, cum = 0, nearest = 0;
            cards.forEach(function(c,i){
                if(Math.abs(cum-pos) < Math.abs((cum-pos)-(-curX-cum))) nearest = i;
                if(cum+c.offsetWidth/2 < pos) nearest = i;
                cum += c.offsetWidth + 14;
            });
            
            if (dragging || Math.abs(velX) > 1) {
                // Keep the active index bound strictly inside available cards
                activeIndex = Math.min(total-1, Math.max(0, nearest));
            }
            
            if (counter) {
                counter.innerHTML = String(activeIndex+1).padStart(2,'0') + ' <span>/ ' + String(total).padStart(2,'0') + '</span>';
            }
        }
        
        function momentum(){
            velX *= 0.93;
            if(Math.abs(velX) < 0.4){ velX = 0; return; }
            applyX(curX + velX, !0);
            raf = requestAnimationFrame(momentum);
        }
